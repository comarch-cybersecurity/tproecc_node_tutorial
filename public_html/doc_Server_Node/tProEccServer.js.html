'use strict';
/**
	* @file <b>TProEcc</b> class handles server side functionalities that serves mainly verification purposes. It consists of following methods:
	* <br> - decryption and verification of digital signature,
	* <br> - public key validation,
	* <br> - pseudo-random number generator.
*/
const EC_lib = require('elliptic').ec;
const crypto_lib = require('crypto');
const ec = new EC_lib('p256');
/**
	* @class
	* @classdesc Simple API, which provides server-side functionalities for verification of operations performed by tPro Ecc token.
*/
class TProEcc {
    /**
	 *  Check whether public key is valid.
	 *  @function validatePublicKey
     *  @param {string} publicKeyHex - ECC public key string representation (hex digits)
     *  @return {{publicKey:{KeyPair}, err:string}}
     */
    validatePublicKey(publicKeyHex) {
        var publicKey;
        try {
            //decode ecc public key from hex representation
            publicKey = ec.keyFromPublic(publicKeyHex, 'hex');

            // check if decoded public key is valid
            var validationResult = publicKey.validate();
            if (validationResult.result === false)
                return {
                    publicKey: null,
                    err: validateResult.reason
                };
        } catch (err) {
            return {
                publicKey: null,
                err: err.message
            };
        }
        return {
            publicKey: publicKey,
            err: null
        };
    }
	/**
	 *  Verify whether signature is valid.
	 *  @function verifySignature
     *  @param {string} publicKeyObj - object which holds key-pair info
	 *  @param {string} messageDigest - hash of the message
	 *  @param {string} signatureOrig - original signature
     *  @return {{result:{string}, err:string}}
     */
    verifySignature(publicKeyObj, messageDigest, signatureOrig) {
        var signature = this.decodeSignature(signatureOrig);
        if (publicKeyObj.constructor.name != 'KeyPair')
            return {
                result: false,
                err: 'Invalid PublicKey object'
            };
        try {
            return {
                result: publicKeyObj.verify(messageDigest, signature),
                err: null
            };
        } catch (err) {
            return {
                result: false,
                err: err.message
            };
        }
    }
	/**
	 *  Utility function to decode signature.
	 *  @function decodeSignature
     *  @param {string} signHex - hex representation of signature
     *  @return {string} 
     */
    decodeSignature(signHex) {
		console.log("s:"+signHex);
        var pos = 0;
        var signArr = this.hexToArray(signHex);
        if (signArr[pos++] != 0x30) return null;
        var signatureLen = signArr[pos++];
        var newSignHex = signHex.substring(0, signatureLen*2 + 4);
		return newSignHex;
        console.log("s:"+newSignHex);
    }
	/**
	 *  Utility function to put hex represented string into array.
	 *  @function hexToArray
     *  @param {string} hexString - hex string to be put into array
     *  @return {array} value
     */
	hexToArray(hexString) {
        var value = [];
        for (var i = 0; i < hexString.length; i += 2) {
            value.push(parseInt(hexString[i] + hexString[i + 1], 16));
        }
        return value;
    }
	/**
	 *  Utility function to generate random number.
	 *  @function rand
     *  @return {string} 
     */
	rand() {
        return crypto_lib.randomBytes(32).toString('hex');
    }
}
module.exports = TProEcc;
